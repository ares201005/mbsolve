51 static void
52 init_fdtd_simulation(
53  std::shared_ptr<const device> dev,
54  std::shared_ptr<scenario> scen,
55  real courant = 1)
56 {
57  if (scen->get_num_gridpoints() > 0) {
58  /* speed of light (use smallest value of relative permittivities) */
59  real velocity =
60  1.0 / sqrt(MU0 * EPS0 * dev->get_minimum_permittivity());
61 
62  /* get number of grid points */
63  unsigned int n_x = scen->get_num_gridpoints();
64 
65  /* grid point size */
66  real d_x = dev->get_length() / (n_x - 1);
67  scen->set_gridpoint_size(d_x);
68 
69  /* time step size */
70  real d_t = courant * d_x / velocity;
71 
72  /* number of time steps */
73  unsigned int n_t = ceil(scen->get_endtime() / d_t) + 1;
74  scen->set_num_timesteps(n_t);
75 
76  /* re-adjust time step size in order to fit number of time steps */
77  d_t = scen->get_endtime() / (n_t - 1);
78  scen->set_timestep_size(d_t);
79 
80  } else {
81  throw std::invalid_argument("Invalid scenario.");
82  }
83 }

91 class sim_constants_fdtd
92 {
93 public:
94  /* prefactor for current electric field in Ampere's law (update equation
95  * for electric field).
96  */
97  real fac_a;
98 
99  /* prefactor for the \partial_x H_y and \partial_t P_z (update equation
100  * for electric field)
101  */
102  real fac_b;
103 
104  /* prefactor for \partial_x E_z (update equation for magnetic field) */

118 static sim_constants_fdtd
119 get_fdtd_constants(
120  std::shared_ptr<const device> dev,
121  std::shared_ptr<const scenario> scen,
122  std::shared_ptr<const material> mat)
123 {
124  sim_constants_fdtd sc;
125 
126  /* convert loss term to conductivity */
127  real sigma = sqrt(
128  EPS0 * mat->get_rel_permittivity() /
129  (MU0 * mat->get_rel_permeability())) *
130  mat->get_losses() * 2.0;
131 
132  real dt = scen->get_timestep_size();
133  real epsilon = EPS0 * mat->get_rel_permittivity();
134  real x = (sigma * dt) / (2.0 * epsilon);
135 
136  /* prefactor a */
137  sc.fac_a = (1.0 - x) / (1.0 + x);
138 
139  /* prefactor b */
140  sc.fac_b = dt / epsilon / (1.0 + x);
141 
142  /* prefactor c */
143  sc.fac_c =
144  dt / (MU0 * mat->get_rel_permeability() * scen->get_gridpoint_size());
145 
146  /* overlap factor */
147  sc.gamma = mat->get_overlap_factor();
148 
149  return sc;
150 }
