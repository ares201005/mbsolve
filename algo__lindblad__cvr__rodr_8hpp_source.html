101  static inline void fill_coefficients(sim_constants& sc)
102  {
103  /* get eigenvalues and eigenvectors in real form */
104  Eigen::RealSchur<real_matrix_t> schur(sc.U);
105  real_matrix_t eigenval = schur.matrixT();
106  real_matrix_t eigenvec = schur.matrixU();
107 
108  /* creating sorting order (descending eigenvalues) */
109  std::vector<size_t> perm_idx(vec_len - 1);
110  std::iota(perm_idx.begin(), perm_idx.end(), 1);
111  std::sort(
112  perm_idx.begin(),
113  perm_idx.end(),
114  [&eigenval](size_t i1, size_t i2) {
115  return std::abs(eigenval(i1, i1 - 1)) >
116  std::abs(eigenval(i2, i2 - 1));
117  });
118 
119  /* TODO optimize
120  * ignore eigenvalues = 0
121  * group eigenvalues with multiplicity >= 2
122  */
123  for (int i = 0; i < vec_len / 2; i++) {
124  unsigned int i1 = perm_idx[i];
125 
126  real_matrix_t b = real_matrix_t::Zero();
127  b(i1, i1 - 1) = +1.0;
128  b(i1 - 1, i1) = -1.0;
129 
130  sc.coeff_1[i] = eigenvec * b * eigenvec.transpose();
131  sc.coeff_2[i] = eigenvec * b * b * eigenvec.transpose();
132  sc.theta[i] = eigenval(i1, i1 - 1);
133  std::cout << "theta: " << std::endl << sc.theta[i] << std::endl;
134  }
135  std::cout << "eigenval: " << eigenval << std::endl;
136  }

139  /*
140  * Determines the simulation constants of the quantum mechanical
141  * description for the generalized Rodrigues formula.
142  */
143  static sim_constants get_qm_constants(
144  std::shared_ptr<const qm_description> qm,
145  real time_step)
146  {
147  sim_constants sc;
148 
149  if (qm) {
150  /* quantum mechanical description present */
151  sc.has_qm = true;
152 
153  /* check whether number of levels matches solver */
154  if (qm->get_num_levels() != num_lvl) {
155  throw std::invalid_argument(
156  "Number of energy levels does not "
157  "match selected solver!");
158  }
159 
160  /* create coherence vector representation */
161  cv_representation cvr(qm);
162 
163  /* polarization constant */
164  sc.M_CP = 0.5 * qm->get_carrier_density();
165 
166  /* time step size */
167  sc.d_t = time_step;
168 
169  /* dissipation term in cvr */
170  real_matrix_t G = cvr.get_relaxation_superop();
171 
172  /* time-independent Hamiltonian contribution in cvr */
173  real_matrix_t M_0 = cvr.get_hamiltonian();
174 
175  /* time-independent part of the Lindblad equation rhs */
176  sc.M = M_0 + G;
177 
178  /* dipole moment Liouvillian in cvr form (matrix) + squared */
179  sc.U = -cvr.get_dipole_operator();
180  sc.U2 = sc.U * sc.U;
181  sc.theta_1 = sqrt(
182  pow(sc.U(0, 1), 2) + pow(sc.U(0, 2), 2) + pow(sc.U(1, 2), 2));
183 
184  /* time-independent propagator, half step */
185  sc.A1 = (sc.M * time_step / 2).exp();
186 
187  /* prepare time-dependent propagator */
188  fill_coefficients(sc);
189 
190  /* dipole moment operator in cvr form (vector) */
191  sc.v = cvr.get_dipole_operator_vec();
192 
193  /* equilibrium terms */
194  sc.d_eq = cvr.get_equilibrium_vec();
195  real eps = std::numeric_limits<real>::epsilon();
196  if (sc.d_eq.isZero(eps)) {
197  sc.d_in = density::Zero();
198  } else {
199  /* solve equation system M * d_in = d_eq */
200  sc.d_in = sc.M.fullPivLu().solve(sc.d_eq);
201  real err = (sc.M * sc.d_in - sc.d_eq).norm() / sc.d_eq.norm();
202  if (err > 1e-3) {
203  throw std::invalid_argument("Time-indepent matrix not "
204  "invertible!");
205  }
206  }
207  } else {
208  /* no quantum mechanical description */
209  sc.has_qm = false;
210  sc.M_CP = 0.0;
211  sc.d_t = 0.0;
212  sc.M = real_matrix_t::Zero();
213  sc.U = real_matrix_t::Zero();
214  sc.U2 = real_matrix_t::Zero();
215  sc.theta_1 = 0.0;
216  sc.A1 = real_matrix_t::Zero();
217  sc.v = density::Zero();
218  sc.d_eq = density::Zero();
219  sc.d_in = density::Zero();
220 
221  /* TODO initialize properly?:
222  real_matrix_t coeff_1[num_lvl/2];
223  real_matrix_t coeff_2[num_lvl/2];
224  real theta[num_lvl/2];
225  */
226  }
227 
228  return sc;
229  }

234  static inline void
235  update(const sim_constants& sc, density& d, real e, real* p_t)
236  {
237  if (sc.has_qm) {
238  /* time-indepedent half step */
239  density d1 = sc.A1 * (d + sc.d_in) - sc.d_in;
240 
241  /* determine time-dependent propagator */
242  real_matrix_t A2;
243 
244  if (num_lvl == 2) {
245  /* original Rodrigues' formula */
246  A2 = sin(sc.theta_1 * e * sc.d_t) / sc.theta_1 * sc.U +
247  (1 - cos(sc.theta_1 * e * sc.d_t)) /
248  (sc.theta_1 * sc.theta_1) * sc.U2 +
249  real_matrix_t::Identity();
250  } else {
251  A2 = real_matrix_t::Identity();
252  for (int i = 0; i < vec_len / 2; i++) {
253  /* TODO nolias()? */
254  A2 += sin(sc.theta[i] * e * sc.d_t) * sc.coeff_1[i] +
255  (1 - cos(sc.theta[i] * e * sc.d_t)) * sc.coeff_2[i];
256  }
257  }
258 
259  /* time-dependent full step */
260  density d2 = A2 * d1;
261 
262  /* time-indepedent half step */
263  d = sc.A1 * (d2 + sc.d_in) - sc.d_in;
264 
265  /* update polarization */
266  *p_t = sc.M_CP * sc.v.transpose() * (sc.M * d + sc.d_eq);
267  }
268  }
